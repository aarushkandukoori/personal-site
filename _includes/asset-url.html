{%- comment -%} Emits a URL for a static asset. Uses github_pages_url in production when set so images load from GitHub Pages while the custom domain DNS still points elsewhere. Usage: {% include asset-url.html path='/assets/img/example.png' %} {%- endcomment -%} {%- if jekyll.environment == 'development' -%} {{- include.path | relative_url -}} {%- elsif site.github_pages_url -%} {{- site.github_pages_url | append: include.path -}} {%- else -%} {{- include.path | relative_url -}} {%- endif -%}
